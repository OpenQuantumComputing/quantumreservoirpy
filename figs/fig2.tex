    \draw[#1] (-#4+#2,#3) arc (180:0:#4 and 0.5*#4);
    % \draw (-.5*#3+#1,#2+.5*#3) node[anchor=north west] {0};
    % \draw (-.5*#3+#1,#2-.5*#3) node[anchor=north west] {1};

}
